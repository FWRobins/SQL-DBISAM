/*
get stock thats prices were changed for a supplier on a date
and export each branches list to CSV
*/

SELECT CODE, DESCRIPT, ONHAND, SELLPINC1
INTO "C:\IQRetail\IQEnterprise\Exports\FWRTABLE"
FROM "C:\IQRetail\IQEnterprise\001\Stock.dat"
WHERE REGULAR_SU = 'ZEL001'
AND LASTSYNCDATE > '2020-05-25'
AND ONHAND > 0;

EXPORT TABLE "C:\IQRetail\IQEnterprise\Exports\FWRTABLE"
TO "C:\IQRetail\IQEnterprise\Exports\001.csv"
WITH HEADERS;
